        /* Element Selector */
        body {
            font-family: Arial, sans-serif;
        }

        /* Class Selector */
        .highlight {
            color: white;
            background-color: green;
            padding: 5px;
        }

        /* ID Selector */
        #main-heading {
            color: navy;
            text-align: center;
        }

        /* Group Selector */
        h1, h2 {
            margin-bottom: 10px;
        }

        /* Universal Selector */
        * {
            box-sizing: border-box;
        }
    </style>
